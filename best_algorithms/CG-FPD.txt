import numpy as np
import torch
import torch.nn as nn
import torch.optim as optim
import torch.nn.functional as F
import gymnasium as gym
from typing import Dict
from torch.utils.tensorboard import SummaryWriter
import os
import json
import math
from typing import Optional 


class NewAlgo:
    """
    STRUCTURAL MACRO-MUTATION (directed):
    Rewrites EXACTLY one major internal mechanism: the planning/distillation teacher.

    Observed failure pattern from fitness:
      - Excellent on CartPole (1.0) and strong on Acrobot (~0.80) -> discrete short-horizon OK.
      - Very weak on HalfCheetah (~0.04) and mediocre on LunarLander/MountainCar ->
        continuous long-horizon + dynamics quality issues.

    Diagnosis:
      - Previous SG-SPD uses "repeat the same first-action a0 for H steps" in imagination.
        This collapses long-horizon reasoning in continuous control: repeating a0 is rarely
        a meaningful open-loop plan and makes the teacher uninformative/noisy, especially
        with imperfect learned dynamics/reward/done predictors.

    Mutation:
        CG-FPD: Cross-Entropy Guided Fractal Plan Distillation.
        * Sample candidate action SEQUENCES (length H), not constant a0.
        * Evaluate sequences in latent space with survival gating (kept).
        * Use a small CEM-style refinement (no gradients, no RL) to sharpen the teacher:
            - keep soft-elite weights, fit a diagonal Gaussian over action sequences,
              resample, repeat for a few iters.
        * Distill ONLY the first action distribution/mean toward the refined teacher.
      - This increases teacher signal quality for continuous tasks without TD/Q/PG.

    External API, policy architecture (2x256 + linear head), optimizer (Adam lr=3e-4),
    and action generation rules remain unchanged.
    """

    def __init__(self, env: gym.Env, lr=3e-4, rollout_len=2048,
                 epochs=8, batch_size=256,
                 eval_interval=5000, verbose=1, log_dir="runs/NewAlgo",algo_cfg: Optional[dict] = None,
                 device=torch.device("cuda:0" if torch.cuda.is_available() else "cpu")):

        self.env = env
        self.verbose = verbose
        self.device = device

        self.rollout_len = int(rollout_len)
        self.epochs = int(epochs)
        self.batch_size = int(batch_size)
        self.eval_interval = int(eval_interval)

        # ----------------- Spaces -----------------
        obs_shape = env.observation_space.shape
        assert obs_shape is not None, "Observation space must have a shape."
        self.obs_dim = int(np.prod(obs_shape))

        self.is_discrete = isinstance(env.action_space, gym.spaces.Discrete)
        if self.is_discrete:
            self.act_dim = env.action_space.n
            self.act_low = None
            self.act_high = None
        else:
            self.act_dim = int(np.prod(env.action_space.shape))
            self.act_low = torch.as_tensor(
                env.action_space.low, dtype=torch.float32, device=self.device
            ).view(1, -1)
            self.act_high = torch.as_tensor(
                env.action_space.high, dtype=torch.float32, device=self.device
            ).view(1, -1)

        # ----------------- Observation Normalizer -----------------
        class RunningMeanStd:
            def __init__(self, shape, clip=5.0, epsilon=1e-8):
                self.shape = (int(np.prod(shape)),)
                self.mean = np.zeros(self.shape, dtype=np.float64)
                self.var = np.ones(self.shape, dtype=np.float64)
                self.count = 1e-4
                self.clip = clip
                self.eps = epsilon

            def update(self, x_np: np.ndarray):
                try:
                    x_np = np.asarray(x_np, dtype=np.float64)
                except Exception:
                    return
                if np.any(~np.isfinite(x_np)):
                    x_np = np.nan_to_num(x_np, nan=0.0, posinf=0.0, neginf=0.0)
                if x_np.ndim == 1:
                    x_np = x_np.reshape(1, -1)
                x_np = x_np.reshape(x_np.shape[0], -1)

                b_mean = np.mean(x_np, axis=0)
                b_var = np.var(x_np, axis=0)
                b_count = x_np.shape[0]

                delta = b_mean - self.mean
                tot_count = self.count + b_count
                if tot_count <= 0:
                    return
                new_mean = self.mean + delta * (b_count / tot_count)
                m_a = self.var * self.count
                m_b = b_var * b_count
                M2 = m_a + m_b + np.square(delta) * (self.count * b_count / tot_count)
                new_var = M2 / max(tot_count, 1e-12)

                self.mean, self.var, self.count = new_mean, new_var, tot_count

            def normalize_torch(self, x: torch.Tensor, device: torch.device) -> torch.Tensor:
                mean = torch.as_tensor(self.mean, dtype=torch.float32, device=device).view(1, -1)
                std = torch.sqrt(torch.as_tensor(self.var, dtype=torch.float32, device=device) + self.eps).view(1, -1)
                x_n = (x - mean) / std
                return torch.clamp(x_n, -self.clip, self.clip)

        self.obs_rms = RunningMeanStd(obs_shape, clip=5.0)

        # ----------------- Policy Network (fixed architecture: 2x256 + linear head) -----------------
        class PolicyNet(nn.Module):
            def __init__(self, obs_dim, act_dim, is_discrete, hidden=256):
                super().__init__()
                self.is_discrete = is_discrete
                # Kept SiLU because the supervision signal is a smooth distillation target (soft teacher);
                # sharp activations tend to amplify teacher noise when model-based planning is imperfect.
                self.trunk = nn.Sequential(
                    nn.Linear(obs_dim, hidden), nn.SiLU(),
                    nn.Linear(hidden, hidden), nn.SiLU(),
                )
                self.head = nn.Linear(hidden, act_dim)

            def forward(self, x: torch.Tensor) -> torch.Tensor:
                return self.head(self.trunk(x))

            def features(self, x: torch.Tensor) -> torch.Tensor:
                return self.trunk(x)

        self.policy = PolicyNet(self.obs_dim, self.act_dim, self.is_discrete).to(self.device)

        # EMA target policy (encoder stabilization)
        self.target_policy = PolicyNet(self.obs_dim, self.act_dim, self.is_discrete).to(self.device)
        self.target_policy.load_state_dict(self.policy.state_dict())
        for p in self.target_policy.parameters():
            p.requires_grad_(False)
        self.tau_ema_policy = 0.01

        # ----------------- Latent Dynamics & Projections -----------------
        class ActionEncoder(nn.Module):
            def __init__(self, act_dim, is_discrete, out_dim=64):
                super().__init__()
                self.is_discrete = is_discrete
                if is_discrete:
                    self.embed = nn.Embedding(act_dim, out_dim)
                else:
                    self.fc = nn.Sequential(
                        nn.Linear(act_dim, 128),
                        nn.LayerNorm(128),
                        nn.SiLU(),
                        nn.Linear(128, out_dim),
                        nn.LayerNorm(out_dim),
                        nn.SiLU(),
                    )

            def forward(self, a: torch.Tensor):
                if self.is_discrete:
                    return self.embed(a.long())
                return self.fc(a)

        self.act_enc = ActionEncoder(self.act_dim, self.is_discrete, out_dim=64).to(self.device)

        class LatentDynamics(nn.Module):
            def __init__(self, feat_dim: int, act_latent_dim: int, hidden=256):
                super().__init__()
                self.net = nn.Sequential(
                    nn.Linear(feat_dim + act_latent_dim, hidden),
                    nn.LayerNorm(hidden),
                    nn.SiLU(),
                    nn.Linear(hidden, hidden),
                    nn.LayerNorm(hidden),
                    nn.SiLU(),
                    nn.Linear(hidden, feat_dim),
                )

            def forward(self, feat: torch.Tensor, a_lat: torch.Tensor) -> torch.Tensor:
                return self.net(torch.cat([feat, a_lat], dim=-1))

        self.latent_dyn = LatentDynamics(256, 64).to(self.device)

        proj_dim = 256

        class Projector(nn.Module):
            def __init__(self, in_dim, out_dim):
                super().__init__()
                self.net = nn.Sequential(
                    nn.Linear(in_dim, out_dim),
                    nn.LayerNorm(out_dim),
                    nn.SiLU(),
                    nn.Linear(out_dim, out_dim),
                    nn.LayerNorm(out_dim),
                    nn.SiLU(),
                )

            def forward(self, x: torch.Tensor) -> torch.Tensor:
                return self.net(x)

        self.proj_enc = Projector(256, proj_dim).to(self.device)
        self.proj_pred = Projector(256, proj_dim).to(self.device)

        # Action temporal prediction head (aux)
        pred_hidden = max(32, self.act_dim // 2)
        self.pred_head = nn.Sequential(
            nn.Linear(self.act_dim, pred_hidden),
            nn.SiLU(),
            nn.Linear(pred_hidden, self.act_dim),
        ).to(self.device)

        # Reward predictor (bounded immediate signal used only for auxiliary + planning score)
        class RewardPredictor(nn.Module):
            def __init__(self, feat_dim=256, act_lat_dim=64, hidden=256):
                super().__init__()
                self.net = nn.Sequential(
                    nn.Linear(feat_dim + act_lat_dim + feat_dim, hidden),
                    nn.LayerNorm(hidden),
                    nn.SiLU(),
                    nn.Linear(hidden, hidden),
                    nn.LayerNorm(hidden),
                    nn.SiLU(),
                    nn.Linear(hidden, 1),
                )

            def forward(self, feat: torch.Tensor, a_lat: torch.Tensor, pred_next_feat: torch.Tensor) -> torch.Tensor:
                return self.net(torch.cat([feat, a_lat, pred_next_feat], dim=-1)).squeeze(-1)

        self.r_pred = RewardPredictor(256, 64, hidden=256).to(self.device)

        # Done predictor (termination awareness for imagined rollouts)
        class DonePredictor(nn.Module):
            def __init__(self, feat_dim=256, act_lat_dim=64, hidden=256):
                super().__init__()
                self.net = nn.Sequential(
                    nn.Linear(feat_dim + act_lat_dim + feat_dim, hidden),
                    nn.LayerNorm(hidden),
                    nn.SiLU(),
                    nn.Linear(hidden, hidden // 2),
                    nn.LayerNorm(hidden // 2),
                    nn.SiLU(),
                    nn.Linear(hidden // 2, 1),
                )

            def forward(self, feat: torch.Tensor, a_lat: torch.Tensor, pred_next_feat: torch.Tensor) -> torch.Tensor:
                return self.net(torch.cat([feat, a_lat, pred_next_feat], dim=-1)).squeeze(-1)

        self.d_pred = DonePredictor(256, 64, hidden=256).to(self.device)

        # ----------------- Optimizer (fixed) -----------------
        self.opt = optim.Adam(
            list(self.policy.parameters()) +
            list(self.act_enc.parameters()) +
            list(self.latent_dyn.parameters()) +
            list(self.proj_enc.parameters()) +
            list(self.proj_pred.parameters()) +
            list(self.pred_head.parameters()) +
            list(self.r_pred.parameters()) +
            list(self.d_pred.parameters()),
            lr=lr
        )

        # ----------------- Rollout Buffer -----------------
        class RolloutBuffer:
            def __init__(self, outer):
                self.outer = outer
                self.reset()

            def add(self, s, a, r, s_next, done):
                self.s.append(s)
                self.a.append(a)
                self.r.append(r)
                self.s_next.append(s_next)
                self.done.append(done)

            def _to_t(self, x, dt):
                return torch.as_tensor(np.array(x), dtype=dt, device=self.outer.device)

            def tensors(self) -> Dict[str, torch.Tensor]:
                out = {
                    "s": self._to_t(self.s, torch.float32).view(len(self.s), -1),
                    "s_next": self._to_t(self.s_next, torch.float32).view(len(self.s_next), -1),
                    "r": self._to_t(self.r, torch.float32).view(-1),
                    "done": self._to_t(self.done, torch.float32).view(-1),
                }
                if self.outer.is_discrete:
                    out["a"] = self._to_t(self.a, torch.long).view(-1)
                else:
                    out["a"] = self._to_t(self.a, torch.float32).view(len(self.a), -1)
                return out

            def reset(self):
                self.s, self.a, self.r, self.s_next, self.done = [], [], [], [], []

        self.buffer = RolloutBuffer(self)

        # ----------------- Logging -----------------
        os.makedirs(log_dir, exist_ok=True)
        os.makedirs(os.path.join(log_dir, "checkpoints"), exist_ok=True)
        self.writer = SummaryWriter(log_dir)
        self.log_dir = log_dir
        self.metrics = {"steps": [], "loss": [], "grad_norm": [], "param_norm": [], "eval_reward": []}

        # ----------------- Robust reward stats -----------------
        self.r_med = 0.0
        self.r_mad = 1.0
        self.r_beta = 0.01
        self.reward_scale = 1.0

        # ----------------- Loss Weights -----------------
        self.w_dyn = 1.0
        self.w_dyn_reward = 0.7
        self.w_contrast = 0.4

        self.w_smooth = 1e-3
        self.w_div = 2e-4
        self.w_reg = 1e-4

        self.anchor_weight = 1e-3
        self.cpc_weight = 1e-3
        self.var_weight = 1e-3

        # Auxiliary predictor weights
        self.rpred_weight = 0.25
        self.done_weight = 0.25

        # ----------------- Planning/distillation (MUTATED MECHANISM: CG-FPD) -----------------
        self.plan_weight = 0.16
        self.plan_horizon = 5                   # slightly longer horizon helps continuous tasks
        self.plan_num_candidates = 32            # more candidates needed for sequence search
        self.plan_temp = 0.7
        self.plan_util_clip = 3.0
        self.plan_cons_weight = 0.06
        self.plan_action_l2 = 0.01

        # survival gating
        self.plan_survive_pow = 1.6
        self.plan_survive_coef = 0.4

        # CEM refinement (major change)
        self.cem_iters = 2
        self.cem_elite_frac = 0.25
        self.cem_min_std = 0.08
        self.cem_max_std = 0.45

        # ----------------- Exploration scheduling -----------------
        self.global_step = 0
        self.exp_horizon = 500_000
        self._disc_tau_max = 1.5
        self._disc_tau_min = 0.3
        self._disc_eps_max = 0.3
        self._disc_eps_min = 0.02
        self._cont_noise_max = 0.30
        self._cont_noise_min = 0.05

        # Gradient clipping
        self.max_grad_norm = 0.8

        # minor (kept for logging consistency)
        self.t_clip_min = 2.0
        self.t_clip_max = 3.5
        if algo_cfg is not None:
            print("setting values in the algorithm")
            for k, v in algo_cfg.items():
                if hasattr(self, k):
                    setattr(self, k, v)

    # ============================================================
    # Helper Functions
    # ============================================================
    def _flatten_obs(self, obs):
        if isinstance(obs, tuple):
            obs = obs[0]
        x = torch.as_tensor(np.asarray(obs), dtype=torch.float32, device=self.device).view(1, -1)
        with torch.no_grad():
            x_n = self.obs_rms.normalize_torch(x, self.device)
        return x_n

    def _normalize_batch_obs(self, x: torch.Tensor) -> torch.Tensor:
        return self.obs_rms.normalize_torch(x, self.device)

    def _explore_schedule(self):
        prog = float(min(1.0, self.global_step / float(self.exp_horizon)))
        tau = self._disc_tau_max + (self._disc_tau_min - self._disc_tau_max) * prog
        eps = self._disc_eps_max + (self._disc_eps_min - self._disc_eps_max) * prog
        sigma = self._cont_noise_max + (self._cont_noise_min - self._cont_noise_max) * prog
        tau = max(tau, 1e-3)
        eps = float(np.clip(eps, 0.0, 1.0))
        sigma = max(sigma, 1e-5)
        return tau, eps, sigma

    def _soft_update_policy_target(self):
        with torch.no_grad():
            for p, tp in zip(self.policy.parameters(), self.target_policy.parameters()):
                tp.data.lerp_(p.data, self.tau_ema_policy)

    def _current_t_clip(self) -> float:
        frac = 1.0 - np.exp(-self.global_step / 3e5)
        return float(self.t_clip_min + (self.t_clip_max - self.t_clip_min) * frac)

    def _update_reward_stats(self, r: torch.Tensor):
        with torch.no_grad():
            if r.numel() == 0:
                return
            b_med = torch.median(r).item()
            b_mad = torch.median(torch.abs(r - b_med)).item() + 1e-6
            self.r_med = (1.0 - self.r_beta) * self.r_med + self.r_beta * b_med
            self.r_mad = (1.0 - self.r_beta) * self.r_mad + self.r_beta * b_mad

    def _robust_reward(self, r: torch.Tensor) -> torch.Tensor:
        self._update_reward_stats(r)
        r_robust = (r - self.r_med) / (1.4826 * self.r_mad + 1e-6)
        return self.reward_scale * torch.tanh(r_robust)

    # ============================================================
    # Saving / Loading
    # ============================================================
    def load(self, path: str):
        checkpoint = torch.load(path, map_location=self.device)
        self.policy.load_state_dict(checkpoint["policy_state"])
        if "target_policy_state" in checkpoint:
            self.target_policy.load_state_dict(checkpoint["target_policy_state"])
        else:
            self.target_policy.load_state_dict(self.policy.state_dict())

        if "act_enc_state" in checkpoint:
            self.act_enc.load_state_dict(checkpoint["act_enc_state"])
        if "latent_dyn_state" in checkpoint:
            self.latent_dyn.load_state_dict(checkpoint["latent_dyn_state"])
        if "proj_enc_state" in checkpoint:
            self.proj_enc.load_state_dict(checkpoint["proj_enc_state"])
        if "proj_pred_state" in checkpoint:
            self.proj_pred.load_state_dict(checkpoint["proj_pred_state"])
        if "pred_head_state" in checkpoint:
            self.pred_head.load_state_dict(checkpoint["pred_head_state"])
        if "r_pred_state" in checkpoint:
            self.r_pred.load_state_dict(checkpoint["r_pred_state"])
        if "d_pred_state" in checkpoint:
            self.d_pred.load_state_dict(checkpoint["d_pred_state"])

        self.opt.load_state_dict(checkpoint["optimizer_state"])
        self.metrics = checkpoint.get("metrics", self.metrics)
        self.global_step = checkpoint.get("global_step", self.global_step)

        obs_rms = checkpoint.get("obs_rms", None)
        if obs_rms is not None:
            self.obs_rms.mean = obs_rms.get("mean", self.obs_rms.mean)
            self.obs_rms.var = obs_rms.get("var", self.obs_rms.var)
            self.obs_rms.count = obs_rms.get("count", self.obs_rms.count)

        self.r_med = checkpoint.get("r_med", self.r_med)
        self.r_mad = checkpoint.get("r_mad", self.r_mad)

        if self.verbose:
            print(f"✅ Loaded checkpoint from {path}")

    # ============================================================
    # Policy Execution
    # ============================================================
    @torch.no_grad()
    def predict(self, obs, deterministic: bool = True):
        x = self._flatten_obs(obs)
        logits = self.policy(x).squeeze(0)

        if self.is_discrete:
            if deterministic:
                action = int(torch.argmax(logits).item())
                return action, None
            tau, eps, _ = self._explore_schedule()
            if torch.rand(1).item() < eps:
                action = int(torch.randint(0, self.act_dim, (1,)).item())
                return action, None
            probs = F.softmax(logits / tau, dim=0)
            probs = torch.clamp(probs, 1e-6, 1.0)
            probs = probs / probs.sum()
            action = int(torch.multinomial(probs, num_samples=1).item())
            return action, None

        base = torch.tanh(logits)
        if deterministic:
            act_t = base
        else:
            _, _, sigma = self._explore_schedule()
            noise = torch.randn_like(base) * sigma
            act_t = torch.clamp(base + noise, -1.0, 1.0)

        act_np = act_t.cpu().numpy().astype(np.float32)
        if self.act_low is not None and self.act_high is not None:
            low = self.act_low.detach().cpu().numpy()
            high = self.act_high.detach().cpu().numpy()
            act_np = low + 0.5 * (act_np + 1.0) * (high - low)
        return act_np.reshape(-1), None

    # ============================================================
    # Rollout Collection
    # ============================================================
    def _collect_rollout(self, total_steps: int) -> int:
        obs, info = self.env.reset()
        self.obs_rms.update(np.asarray(obs, dtype=np.float32).reshape(1, -1))
        steps = 0
        while steps < total_steps:
            a, _ = self.predict(obs, deterministic=False)
            if self.is_discrete:
                env_action = int(a)
                a_buf = int(a)
            else:
                env_action = np.asarray(a, dtype=np.float32).reshape(-1)
                a_buf = env_action

            next_obs, r, terminated, truncated, info = self.env.step(env_action)
            done = bool(terminated or truncated)
            self.obs_rms.update(np.asarray(next_obs, dtype=np.float32).reshape(1, -1))

            r_val = float(np.asarray(r).reshape(-1)[0])
            self.buffer.add(obs, a_buf, r_val, next_obs, float(done))

            obs = next_obs
            steps += 1
            self.global_step += 1

            if done:
                obs, info = self.env.reset()
                self.obs_rms.update(np.asarray(obs, dtype=np.float32).reshape(1, -1))

        return steps

    # ============================================================
    # Core Loss
    # ============================================================
    def compute_loss(self, batch: Dict[str, torch.Tensor]) -> torch.Tensor:
        eps = 1e-8
        s = batch["s"].to(self.device)
        s_next = batch["s_next"].to(self.device)
        r = batch["r"].to(self.device).view(-1)
        done = torch.clamp(batch["done"].to(self.device).view(-1), 0.0, 1.0)
        B = s.shape[0]
        if B == 0:
            return torch.zeros((), dtype=torch.float32, device=self.device)

        # Normalize observations
        s_n = self._normalize_batch_obs(s)
        s_next_n = self._normalize_batch_obs(s_next)

        # Bounded immediate reward proxy (never used as return/TD)
        r_t = self._robust_reward(r)          # [-1,1]
        r_t = r_t * (1.0 - done)              # mask terminal transitions for predictors

        # Features
        feat = self.policy.features(s_n)      # [B,256]
        feat_next = self.policy.features(s_next_n)

        with torch.no_grad():
            feat_t = self.target_policy.features(s_n)
            feat_next_t = self.target_policy.features(s_next_n)

        # Action handling
        if self.is_discrete:
            a_exec = batch["a"].to(self.device).view(-1)
        else:
            a_exec = batch["a"].to(self.device)
            if self.act_low is not None and self.act_high is not None:
                low, high = self.act_low, self.act_high
                a_exec = 2.0 * (a_exec - low) / (high - low + eps) - 1.0
            a_exec = torch.clamp(a_exec, -1.0, 1.0)

        # ===================== 1) Latent dynamics + reward-aligned target =====================
        a_lat_exec = self.act_enc(a_exec if not self.is_discrete else a_exec)
        pred_next_feat = self.latent_dyn(feat, a_lat_exec)

        mask_not_done = (1.0 - done).view(-1, 1)

        L_dyn_raw = F.smooth_l1_loss(
            pred_next_feat * mask_not_done, feat_next_t * mask_not_done, reduction="none"
        ).mean(dim=1)
        L_dyn = self.w_dyn * L_dyn_raw.mean()

        sig = torch.tanh(r_t)
        sig_u = sig.view(-1, 1)
        w_sample = torch.clamp(torch.abs(sig).detach() + 0.05, 0.05, 1.0)

        delta_feat = feat_next - feat_next_t
        aligned_target = feat_next_t + sig_u * torch.clamp(delta_feat, -1.0, 1.0)
        L_dyn_reward_raw = F.smooth_l1_loss(
            pred_next_feat * mask_not_done, aligned_target * mask_not_done, reduction="none"
        ).mean(dim=1)
        L_dyn_reward = self.w_dyn_reward * (w_sample * L_dyn_reward_raw).mean()

        # Contrastive stabilization (binary NCE)
        z_q = F.normalize(self.proj_pred(pred_next_feat.detach()), dim=-1)
        z_k = F.normalize(self.proj_enc(feat_next_t.detach()), dim=-1)
        idx = torch.randperm(B, device=self.device)
        z_neg = z_k[idx]
        pos_logit = (z_q * z_k).sum(dim=-1, keepdim=True) / 0.1
        neg_logit = (z_q * z_neg).sum(dim=-1, keepdim=True) / 0.1
        logits_nc = torch.cat([pos_logit, neg_logit], dim=1)
        labels_nc = torch.zeros(B, dtype=torch.long, device=self.device)
        L_contrast = self.w_contrast * F.cross_entropy(logits_nc, labels_nc)

        # ===================== 2) Policy auxiliaries (smoothness/diversity/reg) =====================
        logits_cur = self.policy(s_n)
        logits_next = self.policy(s_next_n)

        if self.is_discrete:
            tau_train = 0.8
            mu_s = F.softmax(logits_cur / tau_train, dim=-1)
            intent = mu_s
            intent_next = F.softmax(logits_next / tau_train, dim=-1)
        else:
            mu_s = torch.tanh(logits_cur)
            intent = mu_s
            intent_next = torch.tanh(logits_next)

        pred_intent_next = self.pred_head(intent)
        L_smooth = self.w_smooth * (0.5 * (pred_intent_next - intent_next.detach()).pow(2) * mask_not_done).mean()

        z_act = logits_cur - logits_cur.mean(dim=0, keepdim=True)
        cov = (z_act.t() @ z_act) / (float(B) + eps)
        I = torch.eye(self.act_dim, device=self.device)
        L_div = self.w_div * (cov - I).pow(2).mean()

        L_reg = self.w_reg * 0.5 * (logits_cur.pow(2).mean() + logits_next.pow(2).mean())

        # ===================== 3) EMA anchor + CPC-ish action prediction + variance floor =====================
        with torch.no_grad():
            logits_t = self.target_policy(s_n)
            logits_next_t = self.target_policy(s_next_n)

        if self.is_discrete:
            mu_s_t = F.softmax(logits_t / 0.8, dim=-1)
            target_mu_next = F.softmax(logits_next_t / 0.8, dim=-1)
        else:
            mu_s_t = torch.tanh(logits_t)
            target_mu_next = torch.tanh(logits_next_t)

        L_anchor = self.anchor_weight * (mu_s - mu_s_t).pow(2).mean()

        pred_mu_next = self.pred_head(mu_s)
        L_cpc = self.cpc_weight * 0.5 * (((pred_mu_next - target_mu_next) ** 2) * (1.0 - done).view(-1, 1)).mean()

        out_std = mu_s.std(dim=0)
        L_var = self.var_weight * torch.relu(0.2 - out_std).mean()

        # ===================== 4) Auxiliary predictors: reward + done =====================
        r_hat = torch.tanh(self.r_pred(feat.detach(), a_lat_exec.detach(), pred_next_feat.detach()))
        L_rpred = self.rpred_weight * F.smooth_l1_loss(r_hat, r_t.detach())

        d_logit = self.d_pred(feat.detach(), a_lat_exec.detach(), pred_next_feat.detach())
        pos = done.sum()
        neg = done.numel() - pos
        pos_weight = (neg / (pos + 1e-6)).clamp(min=0.2, max=5.0)
        L_dpred = self.done_weight * F.binary_cross_entropy_with_logits(d_logit, done, pos_weight=pos_weight)

        # ===================== 5) MUTATED: CG-FPD (CEM-refined sequence teacher -> first-action distillation) =====================
        H = int(self.plan_horizon)
        C = int(self.plan_num_candidates)

        with torch.no_grad():
            # Base proposal distribution for sequences
            if self.is_discrete:
                # Start from current policy distribution at s for a0; later steps are broad uniform.
                probs0 = F.softmax(logits_cur / 0.9, dim=-1)
                probs0 = torch.clamp(probs0, 1e-6, 1.0)
                probs0 = probs0 / probs0.sum(dim=-1, keepdim=True)
            else:
                _, _, sigma = self._explore_schedule()
                sigma0 = float(np.clip(0.7 * sigma + 0.10, 0.10, 0.35))

                mean_seq = mu_s.detach().unsqueeze(1).unsqueeze(1).expand(B, 1, H, self.act_dim).contiguous()
                std_seq = torch.full((B, 1, H, self.act_dim), sigma0, device=self.device)

            # Helper: evaluate a batch of sequences a_seq [B,C,H,(act_dim)] or [B,C,H] for discrete
            def eval_sequences(a_seq):
                feat_roll = feat_t.unsqueeze(1).expand(-1, a_seq.shape[1], -1).contiguous()  # [B,C,256]
                pref = torch.zeros((B, a_seq.shape[1]), device=self.device)
                surv = torch.ones((B, a_seq.shape[1]), device=self.device)
                cons_pen = torch.zeros((B, a_seq.shape[1]), device=self.device)

                for t in range(H):
                    if self.is_discrete:
                        a_t = a_seq[:, :, t].reshape(-1)
                        a_lat = self.act_enc(a_t).view(B, -1, 64)
                    else:
                        a_t = a_seq[:, :, t, :].reshape(-1, self.act_dim)
                        a_lat = self.act_enc(a_t).view(B, -1, 64)

                    pred_next = self.latent_dyn(
                        feat_roll.reshape(-1, 256),
                        a_lat.reshape(-1, 64)
                    ).view(B, -1, 256)

                    r_step = torch.tanh(
                        self.r_pred(
                            feat_roll.reshape(-1, 256),
                            a_lat.reshape(-1, 64),
                            pred_next.reshape(-1, 256)
                        ).view(B, -1)
                    )

                    d_step_logit = self.d_pred(
                        feat_roll.reshape(-1, 256),
                        a_lat.reshape(-1, 64),
                        pred_next.reshape(-1, 256)
                    ).view(B, -1)
                    p_done = torch.sigmoid(d_step_logit).clamp(0.0, 1.0)

                    util = torch.clamp(
                        r_step + self.plan_survive_coef * (1.0 - p_done),
                        -self.plan_util_clip, self.plan_util_clip
                    )
                    gate = (1.0 - p_done).pow(self.plan_survive_pow)
                    surv = surv * gate
                    pref = pref + surv * util

                    step_jump = (pred_next - feat_roll).pow(2).mean(dim=-1)
                    cons_pen = cons_pen + torch.clamp(step_jump, 0.0, self.plan_util_clip)

                    feat_roll = pred_next

                score = torch.clamp(pref, -self.plan_util_clip, self.plan_util_clip) - self.plan_cons_weight * cons_pen
                return score

            if self.is_discrete:
                # Discrete: sample sequences (a0 from policy mixture, later from mixture uniform/policy)
                # Keep it simple: no CEM for discrete (already strong), but still use sequences (fixes a0-repeat).
                a0_pol = torch.multinomial(probs0, num_samples=C, replacement=True)  # [B,C]
                a0_rand = torch.randint(0, self.act_dim, (B, C), device=self.device)
                mix0 = (torch.rand((B, C), device=self.device) < 0.7)
                a0 = torch.where(mix0, a0_pol, a0_rand).long()

                # later actions broad uniform to allow multi-step consequences
                a_rest = torch.randint(0, self.act_dim, (B, C, H - 1), device=self.device)
                a_seq = torch.cat([a0.unsqueeze(-1), a_rest], dim=-1)  # [B,C,H]

                score = eval_sequences(a_seq)
                w = F.softmax(score / max(self.plan_temp, 1e-3), dim=1)

                teacher = torch.zeros((B, self.act_dim), device=self.device)
                teacher.scatter_add_(1, a0, w)
                teacher = teacher / (teacher.sum(dim=1, keepdim=True) + 1e-6)

            else:
                # Continuous: CEM refinement over sequences
                # Initial sample
                a_seq = torch.clamp(
                    mean_seq.expand(B, C, H, self.act_dim) +
                    torch.randn((B, C, H, self.act_dim), device=self.device) * std_seq.expand(B, C, H, self.act_dim),
                    -1.0, 1.0
                )

                for _ in range(int(self.cem_iters)):
                    score = eval_sequences(a_seq)  # [B,C]
                    # soft-elite weights
                    temp = max(self.plan_temp, 1e-3)
                    w = F.softmax(score / temp, dim=1).clamp(1e-8, 1.0)  # [B,C]
                    w = w / (w.sum(dim=1, keepdim=True) + 1e-8)

                    # compute weighted mean/std over sequences
                    w_u = w.view(B, C, 1, 1)
                    new_mean = (w_u * a_seq).sum(dim=1, keepdim=True)  # [B,1,H,act]
                    diff = a_seq - new_mean
                    new_var = (w_u * diff.pow(2)).sum(dim=1, keepdim=True)
                    new_std = torch.sqrt(new_var + 1e-6)

                    new_std = torch.clamp(new_std, self.cem_min_std, self.cem_max_std)
                    mean_seq = new_mean
                    std_seq = new_std

                    # resample around refined distribution
                    a_seq = torch.clamp(
                        mean_seq.expand(B, C, H, self.act_dim) +
                        torch.randn((B, C, H, self.act_dim), device=self.device) * std_seq.expand(B, C, H, self.act_dim),
                        -1.0, 1.0
                    )

                # final evaluation for teacher weighting
                score = eval_sequences(a_seq)
                w = F.softmax(score / max(self.plan_temp, 1e-3), dim=1)
                a0_teacher = (w.unsqueeze(-1) * a_seq[:, :, 0, :]).sum(dim=1)  # [B,act]
                teacher = a0_teacher  # continuous teacher is a vector

        # Distill toward teacher (only first action)
        if self.is_discrete:
            student_logp = F.log_softmax(logits_cur / max(self.plan_temp, 1e-3), dim=-1)
            L_plan = -(teacher.detach() * student_logp).sum(dim=-1).mean()
        else:
            L_plan = F.smooth_l1_loss(mu_s, teacher.detach())
            L_plan = L_plan + self.plan_action_l2 * teacher.detach().pow(2).mean()

        L_plan = self.plan_weight * L_plan

        loss = (
            L_dyn + L_dyn_reward + L_contrast +
            L_smooth + L_div + L_reg +
            L_anchor + L_cpc + L_var +
            L_rpred + L_dpred +
            L_plan
        )

        if not torch.isfinite(loss):
            loss = torch.zeros((), dtype=torch.float32, device=self.device)
        return loss

    # ============================================================
    # Optimization
    # ============================================================
    def _update(self):
        data = self.buffer.tensors()
        N = data["s"].shape[0]
        if N == 0:
            return {"loss": 0.0, "grad_norm": 0.0, "param_norm": 0.0}

        all_params = (
            list(self.policy.parameters()) +
            list(self.act_enc.parameters()) +
            list(self.latent_dyn.parameters()) +
            list(self.proj_enc.parameters()) +
            list(self.proj_pred.parameters()) +
            list(self.pred_head.parameters()) +
            list(self.r_pred.parameters()) +
            list(self.d_pred.parameters())
        )

        idx_all = torch.arange(N, device=self.device)
        losses, grad_norms, param_norms = [], [], []

        for _ in range(self.epochs):
            perm = torch.randperm(N, device=self.device) if N > self.batch_size else idx_all
            for start in range(0, N, self.batch_size):
                batch_idx = perm[start:start + self.batch_size]
                if batch_idx.numel() == 0:
                    continue
                batch = {k: v[batch_idx] for k, v in data.items()}

                loss = self.compute_loss(batch)
                if not torch.isfinite(loss):
                    continue

                self.opt.zero_grad(set_to_none=True)
                loss.backward()
                grad_norm_val = float(nn.utils.clip_grad_norm_(all_params, self.max_grad_norm))
                if not np.isfinite(grad_norm_val):
                    self.opt.zero_grad(set_to_none=True)
                    continue

                self.opt.step()
                self._soft_update_policy_target()

                with torch.no_grad():
                    param_vec = torch.nn.utils.parameters_to_vector(self.policy.parameters())
                    param_norm = float(param_vec.norm())

                losses.append(float(loss.item()))
                grad_norms.append(grad_norm_val)
                param_norms.append(param_norm)

        self.buffer.reset()

        if len(losses) == 0:
            return {"loss": 0.0, "grad_norm": 0.0, "param_norm": 0.0}
        return {
            "loss": float(np.mean(losses)),
            "grad_norm": float(np.mean(grad_norms)),
            "param_norm": float(np.mean(param_norms)),
        }

    # ============================================================
    # Evaluation
    # ============================================================
    def evaluate(self, n_episodes=5):
        rewards = []
        for _ in range(n_episodes):
            obs, info = self.env.reset()
            done, truncated = False, False
            ep_rew = 0.0
            while not (done or truncated):
                a, _ = self.predict(obs, deterministic=True)
                a_env = int(a) if self.is_discrete else np.asarray(a, dtype=np.float32).reshape(-1)
                obs, r, terminated, truncated, info = self.env.step(a_env)
                done = bool(terminated)
                ep_rew += float(np.asarray(r).reshape(-1)[0])
            rewards.append(ep_rew)
        return float(np.mean(rewards))

    # ============================================================
    # Training Loop
    # ============================================================
    def learn(self, total_timesteps: int):
        steps = 0
        while steps < total_timesteps:
            steps += self._collect_rollout(self.rollout_len)
            logs = self._update()

            self.writer.add_scalar("train/loss", logs["loss"], steps)
            self.writer.add_scalar("train/grad_norm", logs["grad_norm"], steps)
            self.writer.add_scalar("train/param_norm", logs["param_norm"], steps)
            self.writer.add_scalar("train/t_clip_cur", self._current_t_clip(), steps)
            self.writer.add_scalar("train/r_med", float(self.r_med), steps)
            self.writer.add_scalar("train/r_mad", float(self.r_mad), steps)

            self.metrics["steps"].append(steps)
            self.metrics["loss"].append(logs["loss"])
            self.metrics["grad_norm"].append(logs["grad_norm"])
            self.metrics["param_norm"].append(logs["param_norm"])

            if steps % self.eval_interval < self.rollout_len:
                eval_r = self.evaluate()
                self.metrics["eval_reward"].append(eval_r)
                self.writer.add_scalar("eval/mean_reward", eval_r, steps)
                if self.verbose:
                    print(f"[Eval] Mean reward: {eval_r:.2f}")
        return self
